rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /user/{username} {
      allow read: if true;
      allow write: if request.resource.id == username;
      match /collection/{cardId} {
        allow read: if true;
        allow write: if request.resource.parent.id == username;
      }
    }
  }
}
